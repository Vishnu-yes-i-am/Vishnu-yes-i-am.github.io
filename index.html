Q.1-SPIRAL-ORDER 
def rotate(matrix: List[List[int]]) -> None:
    nmatrix=[]
    columns=len(matrix[0])
    rows=len(matrix)
    for i in range(columns):
        mlist=[]
        for j in range(rows):
            mlist.append(matrix[j][columns-1-i])
        nmatrix.append(mlist)
    matrix.clear()
    for i in range(columns):
        matrix.append(nmatrix[i])
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        res = []
        while len(matrix)>0 and len(matrix[0])>0:
            for i in range(len(matrix[0])):
                res.append(matrix[0][i])
            matrix.remove(matrix[0])
            if len(matrix)!=0:
                rotate(matrix)
        return res
Q.2-NEXT-GREATER-ELEMENT 
class Solution:
    def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]:
        mlist=[]
        for i in range(len(nums1)):
            for j in range(nums2.index(nums1[i]),len(nums2)):
                if nums2[j]<=nums1[i] and j!=len(nums2)-1:
                    continue
                elif nums2[j]<=nums1[i] and j==len(nums2)-1:
                    mlist.append(-1)
                else:
                    mlist.append(nums2[j])
                    break
        return mlist
Q.3-TRAIPING-RAIN-WATER 
class Solution:
    def trap(self, height: List[int]) -> List[int]:
        temp=[]
        sumtemp=0
        if len(height)>1:
            for i in range(len(height)):
                temp.append(height[i])
                sumtemp+=height[i]
            first=height[0]
            last=height[len(height)-1]
            max1=height.index(max(height))
            for i in range(1,max1):
                if height[i]=first: 
                    first=height[i] 
            for i in range(len(height)-2,max1-1,-1):
                if height[i]=last: 
                    last=height[i]                
            sumheight=0            
            for i in range(len(height)):
                sumheight+=height[i]       
            return sumheight-sumtemp
        else:
            return 0
Q.4-PRODUCT-EXCEPT-SHELF 
class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        product=1
        mult=1
        vishnu=[]
        for i in range(len(nums)):                                
            product*=nums[i]    
        for j in range(len(nums)):
            if nums[j]!=0:
                vishnu.append(int(product/nums[j]))
            if nums[j]==0:
                for k in range(0,j):
                    mult*=nums[k]
                for k in range(j+1,len(nums)):
                    mult*=nums[k]
                vishnu.append(int(mult))        
        return vishnu
Q.5-ROTATE-IMAGE 
class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
        nmatrix=[]
        n=len(matrix[0])
        for i in range(n):
            mlist=[]
            for j in range(n):
                mlist.append(matrix[n-1-j][i])
            nmatrix.append(mlist)
        matrix.clear()
        for i in range(n):
            matrix.append(nmatrix[i])
